Gamma <- function(n.obs, phi, seed.nr){
  set.seed(seed.nr)
  Alpha <- ts(numeric(n.obs))
  Beta <- rnorm(n.obs)
  for (i in 2:n.obs)
    Alpha[i] <- phi*Alpha[i-1] + Beta[i]
  return(Alpha)
}

Echo <- function(n.obs, phi1, phi2, seed.nr){
